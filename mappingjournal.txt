create table public.weekly_journal_snapshot (
  id uuid not null default gen_random_uuid (),
  user_id uuid not null,
  week_start date not null,
  week_end date not null,
  total_quiz_attempts integer null default 0,
  avg_score numeric(5, 2) null default 0,
  total_study_seconds integer null default 0,
  most_active_hour integer null,
  unique_category_count integer null default 0,
  unique_bab_count integer null default 0,
  review_bab_count integer null default 0,
  cognitive_profile jsonb null default '{}'::jsonb,
  stability_metrics jsonb null default '{}'::jsonb,
  cognitive_summary jsonb null default '{}'::jsonb,
  bab_durations jsonb null default '{}'::jsonb,
  created_at timestamp with time zone null default (now() AT TIME ZONE 'Asia/Jakarta'::text),
  constraint weekly_journal_snapshot_pkey primary key (id),
  constraint weekly_journal_snapshot_user_id_week_start_key unique (user_id, week_start)
) TABLESPACE pg_default;

create index IF not exists idx_weekly_snapshot_user_week on public.weekly_journal_snapshot using btree (user_id, week_start desc) TABLESPACE pg_default;

create table public.study_attempts (
  id bigint generated by default as identity not null,
  question_id text not null,
  is_correct boolean null,
  score integer null,
  submitted_at timestamp without time zone null default now(),
  session_id text null,
  user_answer text null,
  correct_answer text null,
  user_id uuid null default auth.uid (),
  dimension text null,
  duration_seconds integer null,
  category text null,
  title text null,
  created_at timestamp with time zone null default now(),
  constraint study_attempts_pkey primary key (id),
  constraint study_attempts_user_id_fkey foreign KEY (user_id) references auth.users (id),
  constraint chk_dimension_valid check (
    (
      dimension = any (
        array[
          'reading'::text,
          'vocabulary'::text,
          'reasoning'::text,
          'analogy'::text,
          'memory'::text
        ]
      )
    )
  )
) TABLESPACE pg_default;

create index IF not exists idx_study_attempts_user_dimension on public.study_attempts using btree (user_id, dimension) TABLESPACE pg_default;

create trigger trg_after_insert_attempts
after INSERT on study_attempts for EACH row
execute FUNCTION fn_sync_study_progress ();

create trigger trg_sync_attempts_to_session
after INSERT
or
update on study_attempts for EACH row
execute FUNCTION fn_sync_duration_to_session ();

create table public.user_cognitive_sessions (
  id bigint generated by default as identity not null,
  user_id uuid not null,
  session_at timestamp with time zone null default now(),
  iq_estimated numeric null,
  iq_confidence numeric null,
  iq_class text null,
  cognitive_index numeric null,
  stability_index numeric null,
  speed_stability numeric null,
  accuracy_stability numeric null,
  fatigue_drop numeric null,
  scores jsonb null,
  consistency_signature jsonb null,
  source text null default 'auto'::text,
  iq_final numeric null,
  speed_variance double precision null default 0,
  error_consistency double precision null default 0,
  endurance_index double precision null default 0,
  neuro_type character varying(50) null,
  fatigue_sensitivity numeric(6, 2) null,
  neuro_fingerprint jsonb null default '{}'::jsonb,
  memory_score numeric(5, 2) null default 0,
  reading_score numeric(5, 2) null default 0,
  vocabulary_score numeric(5, 2) null default 0,
  reasoning_score numeric(5, 2) null default 0,
  analogy_score numeric(5, 2) null default 0,
  constraint user_cognitive_sessions_pkey primary key (id),
  constraint user_cognitive_sessions_user_id_fkey foreign KEY (user_id) references auth.users (id),
  constraint sessions_iq_confidence_valid check (
    (
      (iq_confidence >= (0)::numeric)
      and (iq_confidence <= (100)::numeric)
    )
  )
) TABLESPACE pg_default;

create index IF not exists idx_sessions_user_id on public.user_cognitive_sessions using btree (user_id) TABLESPACE pg_default;

create index IF not exists idx_sessions_user_date on public.user_cognitive_sessions using btree (user_id, session_at desc) TABLESPACE pg_default;
